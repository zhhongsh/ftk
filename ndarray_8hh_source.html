139 void ndarray<T>::reshape(const std::vector<size_t> &dims_)
140 {
141  dims = dims_;
142  s.resize(dims.size());
143 
144  for (size_t i = 0; i < nd(); i ++)
145  if (i == 0) s[i] = 1;
146  else s[i] = s[i-1]*dims[i-1];
147 
148  p.resize(s[nd()-1]*dims[nd()-1]);
149 }
